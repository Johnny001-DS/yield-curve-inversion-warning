from sklearn.cluster import KMeans
from sklearn.metrics import silhouette_score

# Standardize shape features (so one spread doesnt dominate)
X_shape = StandardScaler().fit_transform(shape_df.values)

k_list = [3,4,5]
sil = []
for k in k_list:
    km = KMeans(n_clusters=k, n_init=50, random_state=0)
    lbl = km.fit_predict(X_shape)
    s = silhouette_score(X_shape, lbl)
    sil.append((k, s))
    print(f"k={k}  silhouette={s:.3f}")

# Choose the k with highest silhouette for a first pass:
best_k = max(sil, key=lambda t: t[1])[0]
print("Best k (by silhouette):", best_k)

km_final = KMeans(n_clusters=best_k, n_init=100, random_state=0)
labels = km_final.fit_predict(X_shape)

shape_df["cluster"] = labels

from statsmodels.tsa.stattools import adfuller, kpss

df_base = pd.read_csv("Data/interim/yields_baseline.csv", parse_dates=["date"], index_col="date")

def adf_p(s):
    s = s.dropna()
    return adfuller(s, autolag="AIC")[1]

from sklearn.cluster import KMeans
from sklearn.preprocessing import StandardScaler

# Rebuild shape_df to be safe
shape_df = pd.DataFrame(index=df_base.index)
shape_df["s_2y_3m"]  = df_base["y_2y"] - df_base["y_3m"]
shape_df["s_5y_3m"]  = df_base["y_5y"] - df_base["y_3m"]
shape_df["s_10y_3m"] = df_base["y_10y"] - df_base["y_3m"]

X_shape = StandardScaler().fit_transform(shape_df.values)
km = KMeans(n_clusters=3, n_init=100, random_state=0).fit(X_shape)
shape_df["cluster"] = km.labels_

# Map to names by centroid order (steep/normal/flat-inverted)
scaler = StandardScaler().fit(shape_df[["s_2y_3m","s_5y_3m","s_10y_3m"]])
cent = pd.DataFrame(scaler.inverse_transform(km.cluster_centers_),
                    columns=["s_2y_3m","s_5y_3m","s_10y_3m"])
order = np.argsort(cent["s_10y_3m"].values)  # smallest=most inverted
name_map = {order[0]:"flat_inverted", order[1]:"normal", order[2]:"steep"}
shape_df["regime"] = shape_df["cluster"].map(name_map)

plt.title("10Y-3M spread with regime shading")

# 2) Transition matrix
lbl = shape_df["regime"].values
states = ["flat_inverted","normal","steep"]
idx = {s:i for i,s in enumerate(states)}
T = np.zeros((3,3), dtype=float)
for i in range(len(lbl)-1):
    T[idx[lbl[i]], idx[lbl[i+1]]] += 1
T = (T.T / T.sum(1)).T  # row-normalize
pd.DataFrame(T, index=states, columns=states)

# 3) Duration stats (weeks spent in a regime)
durations = {s:[] for s in states}
run = 1
for i in range(1, len(lbl)):
    if lbl[i]==lbl[i-1]:
        run += 1
    else:
        durations[lbl[i-1]].append(run)
        run = 1
durations[lbl[-1]].append(run)

f = features.copy()

# 1) First differences (weekly changes)  more stationary
for c in ["slope_10y_3m","slope_10y_2y","curv_10_5_3"]:
    f[f"d_{c}"] = f[c].diff()

# 2) Rolling volatility (realized) over 4, 13, 26 weeks
for c in ["slope_10y_3m","slope_10y_2y"]:
    for w in [4,13,26]:
        f[f"rv_{c}_{w}"] = f[c].diff().rolling(w).std()

# 3) Momentum / moving averages
for c in ["slope_10y_3m","curv_10_5_3"]:
    for w in [4,13,26]:
        f[f"ma_{c}_{w}"] = f[c].rolling(w).mean()
        f[f"roc_{c}_{w}"] = f[c] - f[c].shift(w)  # rate of change

# 4) Regime features: label, one-hot, and "time in regime"

Error-correction terms (ECT) from Johansen

from statsmodels.tsa.vector_ar.vecm import coint_johansen
Y = df_base[["y_3m","y_2y","y_5y","y_10y"]].dropna()
jres = coint_johansen(Y.values, det_order=1, k_ar_diff=2)

# choose rank r (min of where trace_stat > 95% CV, capped at 3)
trace_stat = jres.lr1; cv95 = jres.cvt[:,1]
r = int((trace_stat > cv95).sum()); r = min(r, 3)

beta = jres.evec[:, :r]                     # cointegrating vectors (columns)
ect = Y.values @ beta                       # ECT_t for each relation
ECT = pd.DataFrame(ect, index=Y.index, columns=[f"ect{i+1}" for i in range(r)])
ECT = (ECT - ECT.rolling(260, min_periods=50).mean()) / (ECT.rolling(260, min_periods=50).std())  # z-score for stability

Distance-to-regime boundaries

In[16]:
from sklearn.preprocessing import StandardScaler
from sklearn.cluster import KMeans
import numpy as np, pandas as pd

X = shape_df[["s_2y_3m","s_5y_3m","s_10y_3m"]].values
sc = StandardScaler().fit(X)
Xz = sc.transform(X)
km = KMeans(n_clusters=3, n_init=200, random_state=0).fit(Xz)
shape_df["cluster"] = km.labels_

# signed "distance to inversion side": negative toward inverted centroid
cent = km.cluster_centers_                      # in z-space
# identify which centroid is most inverted by lowest s_10y_3m in original space
cent_unz = sc.inverse_transform(cent)
inv_idx = np.argmin(cent_unz[:,2])              # s_10y_3m is column 2
norm_idx = np.argsort(cent_unz[:,2])[1]
steep_idx= np.argmax(cent_unz[:,2])

# distances to each centroid
d_all = np.stack([np.linalg.norm(Xz - c, axis=1) for c in cent], axis=1)
shape_df["d_inverted"] = d_all[:, inv_idx]
shape_df["d_normal"]   = d_all[:, norm_idx]
shape_df["d_steep"]    = d_all[:, steep_idx]

# regime score: "signed distance" (negative close to inverted, positive close to steep)
# scale to ~[-1, +1]
num = shape_df["d_inverted"] - shape_df["d_steep"]
den = shape_df[["d_inverted","d_steep"]].max(axis=1).replace(0, np.nan)
shape_df["regime_score"] = -(num / den)
